    const now = new Date();
    const end = now;
    let start: Date;
    switch (period) {
      case 'WEEK':
        start = new Date(now);
        start.setDate(start.getDate() - 7);

      case 'QUARTER':

        start.setMonth(start.getMonth() - 3);
        break;
      case 'MONTH':
      default:
        start = new Date(now);
        start.setMonth(start.getMonth() - 1);
        break;
    }
    return { start, end };
  }
