  ): Promise<void> {
    if (runningJobs.has(jobName)) {
      console.log(`[SCHEDULER] ${jobName} already running, skipping`);
      return;
    }
    runningJobs.add(jobName);
    try {
      await fn();
    } finally {
      runningJobs.delete(jobName);
    }
  }

Wrap every scheduled job function with runJobWithLock.

Add a startup verification function that runs when

  async function runStartupChecks(): Promise<void> {
    console.log('[STARTUP] Running Iron Crest health checks...');
    
    const checks = [
      // Database
      { name: 'Database connection', 
        fn: () => db.execute(sql`SELECT 1`) },
      
      // Required env vars
      { name: 'JWT_SECRET configured', 
        fn: () => { if (!config.auth.jwtSecret) 
          throw new Error('Missing') } },
      
      // Role system
      { name: 'No legacy ADMIN/FOUNDER roles',
        fn: async () => {
          const legacy = await db.select()
            .from(users)
            .where(inArray(users.role, 
              ['ADMIN', 'FOUNDER'] as any[]));
          if (legacy.length > 0) 
            throw new Error(
              `${legacy.length} users have legacy roles`
            );
        }},
      
      // Commission system
      { name: 'Rate cards configured',
        fn: async () => {
          const cards = await storage.getActiveRateCards();
          if (cards.length === 0) 
            console.warn('[STARTUP] WARNING: No active rate cards');
        }},
      
      // Required roles exist
      { name: 'ACCOUNTING user exists',
        fn: async () => {
          const accounting = await storage.getUsersByRole(
            'ACCOUNTING'
          );
          if (accounting.length === 0)
            console.warn(
              '[STARTUP] WARNING: No ACCOUNTING users. ' +
              'Accounting overrides cannot be distributed.'
            );
        }},
      
      { name: 'DIRECTOR user exists',
        fn: async () => {
          const directors = await storage.getUsersByRole(
            'DIRECTOR'
          );
          if (directors.length === 0)
            console.warn(
              '[STARTUP] WARNING: No DIRECTOR users. ' +
              'Director overrides cannot be distributed.'
            );
        }},
      
      // Scheduler
      { name: 'Scheduler running',
        fn: () => { /* verify scheduler is initialized */ }},
    ];
    
    let passed = 0;
    let warned = 0;
    let failed = 0;
    
    for (const check of checks) {
      try {
        await check.fn();
        console.log(`[STARTUP] ✓ ${check.name}`);
        passed++;
      } catch (err: any) {
        console.error(`[STARTUP] ✗ ${check.name}: ${err.message}`);
        failed++;
      }
    }
    
    console.log(
      `[STARTUP] Health check complete: ` +
      `${passed} passed, ${warned} warnings, ${failed} failed`
    );
    
    if (failed > 0) {
      console.error(
        '[STARTUP] Critical checks failed. ' +
        'Review errors above before processing live data.'
      );
      // Do NOT crash the server — warn and continue
      // The app should still start so issues can be debugged
    }
  }

Call runStartupChecks() when the server starts, 
after the database connection is established.
